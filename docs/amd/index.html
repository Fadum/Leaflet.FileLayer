Use Leaflet.FileLayer with AMD


requirejs.config({
    baseUrl: '/',
    paths: {
        leaflet: 'https://cdnjs.cloudflare.com/ajax/libs/leaflet/0.7.7/leaflet'
    }
});

requirejs([
    'leaflet',
    '../../src/leaflet.filelayer.js'
],
function (L) {
    var osm = L.tileLayer('TILES_URL', {
        attribution: 'MY_ATTRIBUTION'
    });
    var map = L.map('map', {
        center: [0, 0],
        zoom: 2
    }).addLayer(osm);
    var style = {
        color: 'red',
        opacity: 1.0,
        fillOpacity: 1.0,
        weight: 2,
        clickable: false
    };
    L.Control.FileLayerLoad.LABEL = 'MY_LABEL';
    L.Control.fileLayerLoad({
        fitBounds: true,
        layerOptions: {
            style: style,
            pointToLayer: function (data, latlng) {
                return L.circleMarker(
                    latlng,
                    { style: style }
                );
            }
        }
    }).addTo(map);
});
